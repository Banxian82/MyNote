\subsubsection{Print}

\begin{table}
	%c   & 输出单个字符 \\
	%d   & 十进制数字 \\
	%u   & unsigned 十进制数字 \\
	%o   & 八进制数字 \\
	%x   & unsigned 十六进制数字 \\
	%f   & 十进制单精度浮点数 \\
	%e   & 科学计数法 \\
	%s   & 字符串 \\
	%%   & 输出字符 %  \\
	%f   & 默认输出6 位十进制数字 \\
	%.nf & 输出小数点后 n 位十进制数字 \\
	%mf  & 输出 m 位数字 \\
	%m.nf & 输出 m 位数字 和小数点后 n 位十进制数字

\end{table}
